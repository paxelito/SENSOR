293  def newTechAssessment(self,tmpTime):
294  # According to the average investment evaluation time (once per year) and the agent wealth and policy aids are decreased
295  tmp_tAids = 0
296  for sngAgent in self.allAgents:
297  polToUpdate = sngAgent.invAssessment(self.allTechs, self.allTechsID,tmpTime,self.allAgents,self.allPolicies)
298  # Decrement the total amount of incentives to aid within the system
299  if polToUpdate[0] > 0:
300  self.allPolicies[polToUpdate[0]].residue -= polToUpdate[1]
301  tmp_tAids += polToUpdate[1]
302  # If incentives are vanished, standard no inc policy is attributed to all technologies using the vanish policy
303  if self.allPolicies[polToUpdate[0]].residue <= 0:
304  for sngTech in self.allTechs:
305  if sngTech.policy == polToUpdate[0]:
306  sngTech.policy = 0
307  self.totAids.append(tmp_tAids)

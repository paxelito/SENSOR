309  polToUpdate = sngAgent.invAssessment(self.allTechs, self.allTechsID,tmpTime,self.allAgents,self.allPolicies, self.agroProfit)

311  if polToUpdate[0] > 0:
312  self.allPolicies[polToUpdate[0]].residue -= polToUpdate[1]
313  tmp_tAids += polToUpdate[1]
314  # If incentives are vanished, standard no inc policy is attributed to all technologies using the vanish policy
315  if self.allPolicies[polToUpdate[0]].residue <= 0:
316  for sngTech in self.allTechs:
317  if sngTech.policy == polToUpdate[0]:
318  sngTech.policy = 0
319  self.totAids.append(tmp_tAids)
